### 4.2 状态定义（State）

```python
from typing import TypedDict, List, Optional
from langgraph.graph import StateGraph

class VideoGenState(TypedDict):
    # 输入
    script_id: str
    script_text: str

    # 剧本解析结果
    characters: List[dict]        # 角色提示词列表
    scenes: List[dict]            # 场景提示词列表
    plot_outline: str             # 剧情大纲

    # 分镜表
    storyboards: List[dict]       # 分镜列表

    # 静态提示词
    static_prompts: List[dict]    # 静态提示词列表

    # 动态提示词
    dynamic_prompts: List[dict]   # 动态提示词列表

    # 生成结果
    static_images: List[dict]     # 静态图片URL列表
    video_clips: List[dict]       # 视频片段URL列表

    # 流程控制
    current_step: str
    status: str                   # running / paused / completed / failed
    error: Optional[str]
    edited_flags: dict            # 各环节是否被手动编辑标记
```

### 4.3 节点实现示例

```python
class ScriptParserNode:
    """剧本解析节点：解析剧本，提取角色和场景"""

    def __init__(self, llm_adapter: LLMAdapter):
        self.llm = llm_adapter

    async def __call__(self, state: VideoGenState) -> dict:
        prompt = PARSE_SCRIPT_PROMPT.format(script=state["script_text"])
        result = await self.llm.ainvoke(prompt, response_format="json")

        return {
            "characters": result["characters"],
            "scenes": result["scenes"],
            "plot_outline": result["plot_outline"],
            "current_step": "parse_script_completed"
        }


class StoryboardGeneratorNode:
    """分镜生成节点：结合好莱坞导演手法生成分镜表"""

    def __init__(self, llm_adapter: LLMAdapter):
        self.llm = llm_adapter

    async def __call__(self, state: VideoGenState) -> dict:
        prompt = STORYBOARD_PROMPT.format(
            script=state["script_text"],
            characters=state["characters"],
            scenes=state["scenes"]
        )
        result = await self.llm.ainvoke(prompt, response_format="json")

        return {
            "storyboards": result["storyboards"],
            "current_step": "gen_storyboard_completed"
        }
```

### 4.4 工作流构建

```python
def build_video_gen_workflow() -> StateGraph:
    graph = StateGraph(VideoGenState)

    # 添加节点
    graph.add_node("parse_script", ScriptParserNode(llm))
    graph.add_node("gen_character_prompts", CharacterPromptNode(llm))
    graph.add_node("gen_scene_prompts", ScenePromptNode(llm))
    graph.add_node("gen_storyboard", StoryboardGeneratorNode(llm))
    graph.add_node("gen_static_prompts", StaticPromptNode(llm))
    graph.add_node("gen_static_images", StaticImageNode(image_engine))
    graph.add_node("gen_dynamic_prompts", DynamicPromptNode(llm))
    graph.add_node("gen_videos", VideoGenNode(video_engine))

    # 设置入口
    graph.set_entry_point("parse_script")

    # 添加边（顺序执行）
    graph.add_edge("parse_script", "gen_character_prompts")
    graph.add_edge("gen_character_prompts", "gen_scene_prompts")
    graph.add_edge("gen_scene_prompts", "gen_storyboard")
    graph.add_edge("gen_storyboard", "gen_static_prompts")
    graph.add_edge("gen_static_prompts", "gen_static_images")
    graph.add_edge("gen_static_images", "gen_dynamic_prompts")
    graph.add_edge("gen_dynamic_prompts", "gen_videos")
    graph.add_edge("gen_videos", END)

    # 条件边：支持从任意节点重新开始（编辑后重生成）
    graph.add_conditional_edges(
        "check_edits",
        lambda state: state.get("resume_from", "gen_videos"),
        {
            "gen_storyboard": "gen_storyboard",
            "gen_static_prompts": "gen_static_prompts",
            "gen_dynamic_prompts": "gen_dynamic_prompts",
        }
    )

    return graph.compile()
```

```python
from abc import ABC, abstractmethod
from typing import Any, Optional

class LLMAdapter(ABC):
    """LLM适配器抽象基类，统一不同LLM的调用接口"""

    @abstractmethod
    async def ainvoke(
        self,
        prompt: str,
        system_prompt: Optional[str] = None,
        response_format: str = "text",  # text / json
        temperature: float = 0.7,
        max_tokens: int = 4096,
        **kwargs
    ) -> Any:
        """异步调用LLM"""
        pass

    @abstractmethod
    async def test_connection(self) -> bool:
        """测试连接"""
        pass

    @abstractmethod
    def get_model_info(self) -> dict:
        """获取模型信息"""
        pass
```

#### 6.2.1 OpenAI API 适配器

```python
from langchain_openai import ChatOpenAI

class OpenAIAdapter(LLMAdapter):
    def __init__(self, api_key: str, model: str = "gpt-4o", base_url: str = None):
        self.llm = ChatOpenAI(
            model=model,
            api_key=api_key,
            base_url=base_url,
            temperature=0.7,
        )

    async def ainvoke(self, prompt, system_prompt=None, response_format="text", **kwargs):
        messages = []
        if system_prompt:
            messages.append(SystemMessage(content=system_prompt))
        messages.append(HumanMessage(content=prompt))

        if response_format == "json":
            self.llm = self.llm.bind(response_format={"type": "json_object"})

        result = await self.llm.ainvoke(messages)
        return json.loads(result.content) if response_format == "json" else result.content
```

#### 6.2.2 Ollama 本地适配器

```python
from langchain_ollama import ChatOllama

class OllamaAdapter(LLMAdapter):
    def __init__(self, model: str = "llama3.1:8b", base_url: str = "http://localhost:11434"):
        self.llm = ChatOllama(
            model=model,
            base_url=base_url,
            temperature=0.7,
        )
        self.model_name = model

    async def ainvoke(self, prompt, system_prompt=None, response_format="text", **kwargs):
        messages = []
        if system_prompt:
            messages.append(SystemMessage(content=system_prompt))
        messages.append(HumanMessage(content=prompt))

        result = await self.llm.ainvoke(messages)

        if response_format == "json":
            # Ollama需要手动解析JSON
            return self._extract_json(result.content)
        return result.content

    def _extract_json(self, text: str) -> dict:
        """从文本中提取JSON（Ollama模型可能输出额外文本）"""
        import re
        json_match = re.search(r'\{.*\}', text, re.DOTALL)
        if json_match:
            return json.loads(json_match.group())
        raise ValueError("Failed to extract JSON from LLM response")
```

```python
class ComfyUIImageEngine(ImageEngine):
    def __init__(self, server_url: str = "http://localhost:8188"):
        self.server_url = server_url
        self.client = ComfyUIClient(server_url)

    async def generate(self, prompt, negative_prompt="", width=1024, height=576,
                       seed=-1, character_ref_image=None, **kwargs):
        # 加载工作流模板
        workflow = load_workflow("workflows/txt2img.json")

        # 填充参数
        workflow["6"]["inputs"]["text"] = prompt
        workflow["7"]["inputs"]["text"] = negative_prompt
        workflow["5"]["inputs"]["width"] = width
        workflow["5"]["inputs"]["height"] = height
        if seed >= 0:
            workflow["3"]["inputs"]["seed"] = seed

        # 角色一致性：加载参考图到IPAdapter
        if character_ref_image:
            workflow = self._inject_character_ref(workflow, character_ref_image)

        # 提交并轮询
        prompt_id = await self.client.queue_prompt(workflow)
        result = await self.client.wait_for_completion(prompt_id)

        return {
            "image_url": f"{self.server_url}/view?filename={result['filename']}",
            "seed": seed,
            "metadata": result
        }
```

```python
class ComfyUIVideoEngine(VideoEngine):
    async def generate(self, prompt, negative_prompt="", static_image_url=None,
                       duration=4.0, motion_config=None, **kwargs):
        workflow = load_workflow("workflows/img2video.json")

        # 设置首帧图片
        if static_image_url:
            image_path = await self._download_image(static_image_url)
            workflow["10"]["inputs"]["image"] = image_path

        # 设置运动提示
        workflow["6"]["inputs"]["text"] = prompt
        workflow["7"]["inputs"]["text"] = negative_prompt
        workflow["25"]["inputs"]["length"] = int(duration * 24)  # 帧数

        # 运动配置
        if motion_config:
            workflow = self._apply_motion(workflow, motion_config)

        prompt_id = await self.client.queue_prompt(workflow)
        result = await self.client.wait_for_completion(prompt_id)

        return {
            "video_url": f"{self.server_url}/view?filename={result['filename']}",
            "duration": duration,
            "metadata": result
        }
```

## 8. 角色一致性方案

### 8.1 方案设计

角色一致性是本系统的核心难点，采用**参考图+IPAdapter**方案：

```
剧本解析 -> 生成角色提示词 -> 生成角色参考图（标准定帧）
                                        │
                 ┌──────────────────────┘
                 │
                 ▼
  每个分镜的静态图片生成时：
  1. 加载该镜头涉及角色的参考图
  2. 通过IPAdapter注入角色特征
  3. 结合场景提示词+画面内容生成
  4. 输出保持角色外观一致的图片
```

### 8.2 IPAdapter 工作流

```python
def _inject_character_ref(self, workflow: dict, character_ref_images: list):
    """将角色参考图注入ComfyUI工作流的IPAdapter节点"""

    for i, ref_image in enumerate(character_ref_images):
        # 添加LoadImage节点
        load_node_id = f"ref_load_{i}"
        workflow[load_node_id] = {
            "class_type": "LoadImage",
            "inputs": {"image": ref_image}
        }

        # 连接到IPAdapter节点
        ipadapter_node_id = f"ipadapter_{i}"
        workflow[ipadapter_node_id] = {
            "class_type": "IPAdapterApply",
            "inputs": {
                "model": ["MODEL", 0],
                "image": [load_node_id, 0],
                "weight": 0.85,
                "noise": 0.1
            }
        }

    return workflow
```

### 8.3 参考图生成流程

```
角色提示词生成完成
    │
    ▼
为每个角色生成标准参考图：
  - 正面半身照
  - 侧面照
  - 全身照
    │
    ▼
参考图存储到MinIO
    │
    ▼
后续分镜图片生成时自动加载对应角色参考图
```

## 10. 安全设计

### 10.1 认证与授权

```
用户登录 -> JWT Token签发 -> 请求携带 Authorization: Bearer <token>
                                         │
                                         ▼
                                   中间件验证Token
                                         │
                                    ┌────┴────┐
                                    │         │
                                有效Token  无效Token
                                    │         │
                                    ▼         ▼
                              继续处理    返回401
```

### 10.2 敏感信息保护

| 信息 | 保护方式 |
|------|----------|
| 数据库密码 | 环境变量注入，不入代码库 |
| API Key | 加密存储在数据库，运行时解密 |
| JWT密钥 | 环境变量 |
| MinIO密钥 | 环境变量 |
| 用户密码 | bcrypt哈希存储 |

### 10.3 API安全

- 所有接口需JWT认证（除登录接口）
- 请求频率限制（Rate Limiting）：100次/分钟
- 文件上传大小限制：图片10MB，视频500MB
- CORS白名单配置
- SQL注入防护（SQLAlchemy参数化查询）
